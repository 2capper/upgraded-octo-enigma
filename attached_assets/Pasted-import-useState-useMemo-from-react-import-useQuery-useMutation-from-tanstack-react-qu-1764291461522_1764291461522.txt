import { useState, useMemo } from "react";
import { useQuery, useMutation } from "@tanstack/react-query";
import { format, addMinutes, parse, isWithinInterval } from "date-fns";
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { Badge } from "@/components/ui/badge";
import { Loader2, Calendar, Lock, Unlock, Trash2, Plus } from "lucide-react";
import { apiRequest, queryClient } from "@/lib/queryClient";
import { useToast } from "@/hooks/use-toast";
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from "@/components/ui/dialog";
import { Label } from "@/components/ui/label";
import { Input } from "@/components/ui/input";
import type { Diamond, TournamentDiamondAllocation, AgeDivision } from "@shared/schema";

interface FieldManagerProps {
  tournamentId: string;
  startDate: string;
  endDate: string;
  diamonds: Diamond[];
  ageDivisions: AgeDivision[];
}

export function FieldAllocationManager({ tournamentId, startDate, endDate, diamonds, ageDivisions }: FieldManagerProps) {
  const { toast } = useToast();
  const [selectedDate, setSelectedDate] = useState(startDate);
  const [isDialogOpen, setIsDialogOpen] = useState(false);
  const [selectedDiamond, setSelectedDiamond] = useState<string | null>(null);
  
  // New Allocation Form State
  const [startTime, setStartTime] = useState("08:00");
  const [endTime, setEndTime] = useState("22:00");
  const [selectedDivision, setSelectedDivision] = useState<string>("any");

  // Fetch Allocations
  const { data: allocations = [], isLoading } = useQuery<TournamentDiamondAllocation[]>({
    queryKey: [`/api/tournaments/${tournamentId}/allocations`],
  });

  // Filter allocations for current view
  const dayAllocations = useMemo(() => {
    return allocations.filter(a => a.date === selectedDate);
  }, [allocations, selectedDate]);

  // Mutations
  const createMutation = useMutation({
    mutationFn: async () => {
      if (!selectedDiamond) throw new Error("No diamond selected");
      const payload = {
        tournamentId,
        diamondId: selectedDiamond,
        date: selectedDate,
        startTime,
        endTime,
        divisionId: selectedDivision === "any" ? null : selectedDivision,
      };
      return apiRequest("POST", "/api/allocations", payload);
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: [`/api/tournaments/${tournamentId}/allocations`] });
      toast({ title: "Time Block Reserved", description: "Diamond allocation saved." });
      setIsDialogOpen(false);
    },
    onError: (err: any) => toast({ title: "Error", description: err.message, variant: "destructive" })
  });

  const deleteMutation = useMutation({
    mutationFn: async (id: string) => {
      return apiRequest("DELETE", `/api/allocations/${id}`);
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: [`/api/tournaments/${tournamentId}/allocations`] });
      toast({ title: "Removed", description: "Allocation block deleted." });
    }
  });

  // Generate Date Options
  const dateOptions = useMemo(() => {
    const dates = [];
    let curr = new Date(startDate + "T00:00:00"); // Force local time interpretation
    const end = new Date(endDate + "T00:00:00");
    
    while (curr <= end) {
      dates.push(format(curr, "yyyy-MM-dd"));
      curr.setDate(curr.getDate() + 1);
    }
    return dates;
  }, [startDate, endDate]);

  const handleAddClick = (diamondId: string) => {
    setSelectedDiamond(diamondId);
    setStartTime("08:00");
    setEndTime("22:00");
    setSelectedDivision("any");
    setIsDialogOpen(true);
  };

  return (
    <div className="space-y-6">
      {/* Controls */}
      <div className="flex items-center justify-between bg-white p-4 rounded-lg border shadow-sm">
        <div className="flex items-center gap-4">
          <div className="flex items-center gap-2 text-sm font-medium text-gray-700">
            <Calendar className="h-4 w-4" /> Viewing Date:
          </div>
          <Select value={selectedDate} onValueChange={setSelectedDate}>
            <SelectTrigger className="w-[180px]">
              <SelectValue />
            </SelectTrigger>
            <SelectContent>
              {dateOptions.map(date => (
                <SelectItem key={date} value={date}>
                  {format(new Date(date + "T00:00:00"), "EEE, MMM d")}
                </SelectItem>
              ))}
            </SelectContent>
          </Select>
        </div>
        <div className="text-sm text-gray-500">
          {dayAllocations.length} blocks allocated
        </div>
      </div>

      {/* Visual Grid */}
      <div className="grid gap-4">
        {diamonds.map(diamond => {
          const diamondAllocations = dayAllocations.filter(a => a.diamondId === diamond.id);
          // Sort by start time
          diamondAllocations.sort((a, b) => a.startTime.localeCompare(b.startTime));

          return (
            <Card key={diamond.id} className="overflow-hidden">
              <div className="flex flex-col md:flex-row border-l-4 border-l-[var(--brand-primary)]">
                {/* Diamond Info */}
                <div className="p-4 md:w-48 bg-gray-50 border-r flex flex-col justify-center">
                  <h3 className="font-bold text-gray-900">{diamond.name}</h3>
                  <p className="text-xs text-gray-500">{diamond.location || "No location set"}</p>
                  <Button 
                    variant="outline" 
                    size="sm" 
                    className="mt-2 w-full text-xs"
                    onClick={() => handleAddClick(diamond.id)}
                  >
                    <Plus className="h-3 w-3 mr-1" /> Add Block
                  </Button>
                </div>

                {/* Timeline Visualization */}
                <div className="flex-1 p-4 bg-white min-h-[100px] relative">
                  {diamondAllocations.length === 0 ? (
                    <div className="h-full flex items-center justify-center text-gray-400 text-sm italic border-2 border-dashed rounded-md">
                      No time allocated for this day. Click "Add Block" to reserve.
                    </div>
                  ) : (
                    <div className="space-y-2">
                      {diamondAllocations.map(alloc => {
                        const divisionName = ageDivisions.find(d => d.id === alloc.divisionId)?.name || "All Divisions";
                        return (
                          <div 
                            key={alloc.id} 
                            className="flex items-center justify-between p-3 rounded-md bg-green-50 border border-green-200"
                          >
                            <div className="flex items-center gap-3">
                              <Badge className="bg-green-600 hover:bg-green-700">
                                {alloc.startTime} - {alloc.endTime}
                              </Badge>
                              <span className="text-sm font-medium text-gray-900">
                                {alloc.divisionId ? `Reserved for ${divisionName}` : "Open to All Divisions"}
                              </span>
                            </div>
                            <Button 
                              variant="ghost" 
                              size="sm" 
                              className="text-red-500 hover:text-red-700 hover:bg-red-50 h-8 w-8 p-0"
                              onClick={() => {
                                if (confirm("Release this time block?")) deleteMutation.mutate(alloc.id);
                              }}
                            >
                              <Trash2 className="h-4 w-4" />
                            </Button>
                          </div>
                        );
                      })}
                    </div>
                  )}
                </div>
              </div>
            </Card>
          );
        })}
      </div>

      {/* Add Allocation Dialog */}
      <Dialog open={isDialogOpen} onOpenChange={setIsDialogOpen}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>Reserve Time Block</DialogTitle>
          </DialogHeader>
          <div className="space-y-4 py-4">
            <div className="grid grid-cols-2 gap-4">
              <div className="space-y-2">
                <Label>Start Time</Label>
                <Input type="time" value={startTime} onChange={e => setStartTime(e.target.value)} />
              </div>
              <div className="space-y-2">
                <Label>End Time</Label>
                <Input type="time" value={endTime} onChange={e => setEndTime(e.target.value)} />
              </div>
            </div>
            
            <div className="space-y-2">
              <Label>Restriction (Optional)</Label>
              <Select value={selectedDivision} onValueChange={setSelectedDivision}>
                <SelectTrigger>
                  <SelectValue placeholder="Select Restriction" />
                </SelectTrigger>
                <SelectContent>
                  <SelectItem value="any">Open to All Divisions</SelectItem>
                  {ageDivisions.map(div => (
                    <SelectItem key={div.id} value={div.id}>Only {div.name}</SelectItem>
                  ))}
                </SelectContent>
              </Select>
              <p className="text-xs text-muted-foreground">
                If selected, the scheduler will warn if you try to book a different division here.
              </p>
            </div>
          </div>
          <DialogFooter>
            <Button variant="outline" onClick={() => setIsDialogOpen(false)}>Cancel</Button>
            <Button onClick={() => createMutation.mutate()} disabled={createMutation.isPending}>
              {createMutation.isPending ? <Loader2 className="h-4 w-4 animate-spin mr-2" /> : null}
              Reserve Block
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </div>
  );
}
